# .pkg Format Quick Reference for Developers

## Package Format Comparison

| Aspect | .pkg (Modern) | .nupkg (Legacy) |
|--------|---------------|-----------------|
| **Archive** | Standard ZIP | NuGet ZIP |
| **Metadata** | build-info.yaml (YAML) | *.nuspec (XML) |
| **Signatures** | Embedded in YAML | External .signature.p7s |
| **Installer** | sbin-installer | Chocolatey |
| **Scripts** | Native PowerShell | Chocolatey wrappers |
| **Dependencies** | Manual specification | Automatic resolution |

## Core .pkg Structure

```
myapp-1.0.0.pkg
├── build-info.yaml    # Metadata + embedded signature
├── payload/           # Files to install
│   ├── app.exe
│   └── config.ini
└── scripts/           # Optional installation scripts
    ├── preinstall.ps1
    └── postinstall.ps1
```

## build-info.yaml Template

### Basic Package Metadata
```yaml
product:
  name: "My Application"
  version: "1.0.0"
  identifier: "com.company.myapp"
  developer: "Company Name"
  description: "Application description"
  architecture: "x64"

install_location: "C:\\Program Files\\MyApp"  # Copy-type package
# install_location: ""                        # Installer-type package

postinstall_action: "none"  # none, logout, restart
```

### Signature Metadata (Auto-Generated by cimipkg -sign)
```yaml
signature:
  algorithm: "SHA256withRSA"
  certificate:
    subject: "CN=Company, O=Organization"
    thumbprint: "A1B2C3D4..."
    not_before: "2024-01-01T00:00:00Z"
    not_after: "2025-12-31T23:59:59Z"
  package_hash: "sha256:e3b0c44..."
  content_hash: "sha256:a665a45..."
  signed_hash: "6b86b273..."
  timestamp: "2024-12-20T10:30:45Z"
  version: "1.0"
```

## Building Packages

### Create .pkg Package (Default)
```bash
# Basic build
cimipkg -v 1.0.0

# With signing
cimipkg -v 1.0.0 -sign

# Specific certificate
cimipkg -v 1.0.0 -sign -thumbprint "A1B2C3D4..."
```

### Create .nupkg Package (Legacy)
```bash
# Basic .nupkg build
cimipkg -v 1.0.0 --nupkg

# With signing and Intune wrapper
cimipkg -v 1.0.0 --nupkg -sign -intunewin
```

## Package Types

### 1. Copy-Type Package
```yaml
# Files copied to specified location
install_location: "C:\\Program Files\\MyApp"
```

### 2. Installer-Type Package  
```yaml
# Embedded installer executed
install_location: ""  # Empty = installer mode
installer:
  type: "msi"
  silent_args: "/qn"
  exit_codes: [0, 3010]
```

### 3. Script-Only Package
```yaml
# Only scripts executed
install_location: ""  # No files to copy
# Scripts in scripts/ directory run automatically
```

## Code Integration Examples

### Detecting Package Format
```go
func getPackageFormat(packagePath string) string {
    if strings.HasSuffix(strings.ToLower(packagePath), ".pkg") {
        return "pkg"
    } else if strings.HasSuffix(strings.ToLower(packagePath), ".nupkg") {
        return "nupkg"
    }
    return "unknown"
}
```

### Extracting .pkg Metadata
```go
func extractPkgMetadata(packagePath string) (*BuildInfo, error) {
    zipReader, err := zip.OpenReader(packagePath)
    if err != nil {
        return nil, err
    }
    defer zipReader.Close()
    
    for _, file := range zipReader.File {
        if file.Name == "build-info.yaml" {
            rc, err := file.Open()
            if err != nil {
                return nil, err
            }
            defer rc.Close()
            
            data, err := io.ReadAll(rc)
            if err != nil {
                return nil, err
            }
            
            var buildInfo BuildInfo
            if err := yaml.Unmarshal(data, &buildInfo); err != nil {
                return nil, err
            }
            
            return &buildInfo, nil
        }
    }
    
    return nil, fmt.Errorf("build-info.yaml not found in package")
}
```

### Verifying Package Signature
```go
func verifyPkgSignature(packagePath string) (bool, error) {
    buildInfo, err := extractPkgMetadata(packagePath)
    if err != nil {
        return false, err
    }
    
    if buildInfo.Signature == nil {
        return false, fmt.Errorf("package not signed")
    }
    
    // Recalculate package hash
    calculatedHash, err := calculatePackageHash(packagePath)
    if err != nil {
        return false, err
    }
    
    // Verify signature matches calculated hash
    return verifyRSASignature(
        buildInfo.Signature.SignedHash,
        calculatedHash,
        buildInfo.Signature.Certificate.Thumbprint,
    )
}
```

## Installation Integration

### Cimian Client Integration
```go
func installPackage(packagePath string) error {
    format := getPackageFormat(packagePath)
    
    switch format {
    case "pkg":
        return installWithSbinInstaller(packagePath)
    case "nupkg":
        if config.PreferSbinInstaller && sbinInstallerAvailable() {
            return installWithSbinInstaller(packagePath)
        }
        return installWithChocolatey(packagePath)
    default:
        return fmt.Errorf("unsupported package format: %s", format)
    }
}

func installWithSbinInstaller(packagePath string) error {
    cmd := exec.Command("sbin-installer", "--package", packagePath, "--verbose")
    output, err := cmd.CombinedOutput()
    
    if err != nil {
        return fmt.Errorf("sbin-installer failed: %v\nOutput: %s", err, output)
    }
    
    return nil
}
```

## Common Data Structures

### BuildInfo Structure
```go
type BuildInfo struct {
    Product   ProductInfo `yaml:"product"`
    InstallLocation string `yaml:"install_location,omitempty"`
    PostInstallAction string `yaml:"postinstall_action,omitempty"`
    Installer InstallerInfo `yaml:"installer,omitempty"`
    Signature *PackageSignature `yaml:"signature,omitempty"`
}

type ProductInfo struct {
    Name         string `yaml:"name"`
    Version      string `yaml:"version"`
    Identifier   string `yaml:"identifier"`
    Developer    string `yaml:"developer"`
    Description  string `yaml:"description,omitempty"`
    Architecture string `yaml:"architecture,omitempty"`
}

type PackageSignature struct {
    Algorithm    string           `yaml:"algorithm"`
    Certificate  CertificateInfo  `yaml:"certificate"`
    PackageHash  string           `yaml:"package_hash"`
    ContentHash  string           `yaml:"content_hash"`
    SignedHash   string           `yaml:"signed_hash"`
    Timestamp    string           `yaml:"timestamp"`
    Version      string           `yaml:"version"`
}
```

## Testing Helpers

### Create Test Package
```go
func createTestPkgPackage(t *testing.T, name, version string) string {
    tmpDir := t.TempDir()
    pkgPath := filepath.Join(tmpDir, fmt.Sprintf("%s-%s.pkg", name, version))
    
    zipFile, err := os.Create(pkgPath)
    require.NoError(t, err)
    defer zipFile.Close()
    
    zipWriter := zip.NewWriter(zipFile)
    defer zipWriter.Close()
    
    // Add build-info.yaml
    buildInfo := BuildInfo{
        Product: ProductInfo{
            Name:       name,
            Version:    version,
            Identifier: fmt.Sprintf("test.%s", name),
            Developer:  "Test Developer",
        },
    }
    
    buildInfoData, err := yaml.Marshal(buildInfo)
    require.NoError(t, err)
    
    buildInfoFile, err := zipWriter.Create("build-info.yaml")
    require.NoError(t, err)
    _, err = buildInfoFile.Write(buildInfoData)
    require.NoError(t, err)
    
    return pkgPath
}
```

## Migration Helpers

### Format Detection
```go
func detectPackageFormat(packagePath string) (string, error) {
    ext := strings.ToLower(filepath.Ext(packagePath))
    switch ext {
    case ".pkg":
        return "pkg", nil
    case ".nupkg":
        return "nupkg", nil
    default:
        return "", fmt.Errorf("unknown package extension: %s", ext)
    }
}
```

### Dual-Format Processing
```go
func processPackageMetadata(packagePath string) (PackageMetadata, error) {
    format, err := detectPackageFormat(packagePath)
    if err != nil {
        return PackageMetadata{}, err
    }
    
    switch format {
    case "pkg":
        return extractPkgMetadata(packagePath)
    case "nupkg":
        return extractNupkgMetadata(packagePath)
    default:
        return PackageMetadata{}, fmt.Errorf("unsupported format: %s", format)
    }
}
```
